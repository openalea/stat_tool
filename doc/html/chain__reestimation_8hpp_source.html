105template <typename Type>
106ChainReestimation<Type>::ChainReestimation(process_type itype , int inb_state ,
107 int inb_row , bool init_flag)
108
109{
110 int i;
111
112
113 type = itype;
114 nb_state = inb_state;
115 nb_row = inb_row;
116
117 switch (type) {
118 case ORDINARY :
119 initial = new Type[nb_state];
120 break;
121 case EQUILIBRIUM :
122 initial = new Type[nb_row];
123 break;
124 }
125
126 transition = new Type*[nb_row];
127 for (i = 0;i < nb_row;i++) {
128 transition[i] = new Type[nb_state];
129 }
130
131 if (init_flag) {
132 init();
133 }
134}

145template <typename Type>
146void ChainReestimation<Type>::copy(const ChainReestimation<Type> &chain_data)
147
148{
149 int i , j;
150
151
152 nb_state = chain_data.nb_state;
153 nb_row = chain_data.nb_row;
154
155 if (chain_data.initial) {
156 switch (type) {
157
158 case ORDINARY : {
159 initial = new Type[nb_state];
160 for (i = 0;i < nb_state;i++) {
161 initial[i] = chain_data.initial[i];
162 }
163 break;
164 }
165
166 case EQUILIBRIUM : {
167 initial = new Type[nb_row];
168 for (i = 0;i < nb_row;i++) {
169 initial[i] = chain_data.initial[i];
170 }
171 break;
172 }
173 }
174 }
175
176 else {
177 initial = NULL;
178 }
179
180 transition = new Type*[nb_row];
181 for (i = 0;i < nb_row;i++) {
182 transition[i] = new Type[nb_state];
183
184 for (j = 0;j < nb_state;j++) {
185 transition[i][j] = chain_data.transition[i][j];
186 }
187 }
188}
